{
        GET(TechnoClass *, T, ESI);
        if(T->WhatAmI() == abs_Building) {
                T->DestroyGap();
        } else {
                TechnoExt::NeedsRegap = 1;
                FootClass *F = reinterpret_cast<FootClass *>(T);
                CellStruct buffer = F->LastMapCoords;
                F->LastMapCoords = F->CurrentMapCoords;
                F->DestroyGap();
                F->LastMapCoords = buffer;
        }
        return 0x6F6B70;
}

        GET(FootClass *, F, ESI);
